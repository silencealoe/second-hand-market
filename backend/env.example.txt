# ============================================
# 服务器配置
# ============================================
# 服务器端口
PORT=3000

# 服务器基础URL（用于生成文件访问链接）
# 生产环境请修改为实际域名，例如：https://api.yourdomain.com
BASE_URL=http://localhost:3000

# 前端应用URL（用于支付成功后的跳转）
# 生产环境请修改为实际域名，例如：https://yourdomain.com
FRONTEND_URL=http://localhost:5173

# ============================================
# 数据库配置
# ============================================
DB_HOST=localhost
DB_PORT=3306
DB_USERNAME=root
DB_PASSWORD=
DB_DATABASE=second_hand_market

# ============================================
# JWT 配置
# ============================================
# JWT 密钥（生产环境请使用强随机字符串）
JWT_SECRET=your-secret-key-change-in-production

# JWT 过期时间（例如：7d, 24h, 60m）
JWT_EXPIRES_IN=7d

# ============================================
# 支付宝支付配置
# ============================================
# 支付宝应用ID（APPID）
# 在支付宝开放平台创建应用后获取
ALIPAY_APP_ID=your_alipay_app_id

# 支付宝网关地址
# 沙箱环境：https://openapi.alipaydev.com/gateway.do
# 生产环境：https://openapi.alipay.com/gateway.do
ALIPAY_GATEWAY=https://openapi.alipaydev.com/gateway.do

# 应用私钥（RSA2私钥）
# 注意：私钥需要去除头尾的 -----BEGIN PRIVATE KEY----- 和 -----END PRIVATE KEY-----
# 并且需要将换行符替换为 \n，或者直接使用单行格式
# 示例格式：MIIEvQIBADANBgkqhkiG9w0BAQEFAASCBKcwggSjAgEAAoIBAQC...
ALIPAY_PRIVATE_KEY=your_rsa2_private_key

# 支付宝公钥
# 在支付宝开放平台获取支付宝公钥
# 注意：同样需要去除头尾标记，使用单行格式
ALIPAY_PUBLIC_KEY=your_alipay_public_key

# 签名类型（可选，默认为 RSA2）
# 可选值：RSA2, RSA
ALIPAY_SIGN_TYPE=RSA2

# 支付宝异步通知地址
# 支付完成后，支付宝会向此地址发送异步通知
# 请根据实际部署环境修改域名
ALIPAY_NOTIFY_URL=http://localhost:3000/orders/alipay/notify

# 支付宝同步返回地址
# 支付完成后，用户浏览器会跳转到此地址
# 请根据实际部署环境修改域名
ALIPAY_RETURN_URL=http://localhost:3000/orders/alipay/return

# ============================================
# Redis 配置
# ============================================
# Redis 连接地址
# 默认：redis://127.0.0.1:6379
REDIS_URL=redis://127.0.0.1:6379

# ============================================
# 订单配置
# ============================================
# 订单支付超时时间（秒）
# 超过此时间未支付，订单将自动取消
# 默认：15秒
ORDER_PAYMENT_TIMEOUT=15
